- Now create a venv in order to keep various versions & dependencies of various project.
```sh
python -m venv myenv
myenv\Scripts\activate
Set-ExecutionPolicy -Scope Process -ExectionPolicy Bypass
myenv\Scripts\activate
pip list
cd .\water-potability-tracking\
``` 
- Remove unnecessary files from the project if necessary
- Copy the data colletion and processing py files to the src\data folder under water-potability-prection folder, model building, evaution files to src\models folder, params.yml outside src
- In cookie cutter update the requirements.txt with require libraries and run **pip install -r requirements.txt**
- Freeze the requirements.txt file **pip freeze > requirements.txt**
- once all done create dvc pipeline
```sh
git init          # only if not already a git repo
dvc init

dvc stage add --force -n data_collection `
  -d src/data/data_collection.py `
  -o data/raw `
  --params data_collection.test_size `
  python -m src.data.data_collection
# check dvc.yaml file and data/raw
dvc repro # check dvc.lock
dvc dag

dvc stage add --force -n data_process `
  -d src/data/data_processing.py `
  -d data/raw `
  -o data/processed `
  python -m src.data.data_processing


# check dvc.yaml file and data/processed
dvc repro # check dvc.lock
dvc dag

dvc stage add --force -n model_build `
  -d src/model/model_building.py `
  -d src/data/data_collection.py `
  -d data/processed `
  -o model.pkl `
  --params model_building.n_estimators `
  python -m src.model.model_building
# check dvc.yaml file and model.pkl
dvc repro # check dvc.lock
dvc dag
```
- Install required libraries **pip install dvclive** and add live_dvc_exp to get the tracking

```sh
dvc stage add --force -n model_eval `
  -d src/model/model_evaluation.py `
  -d model.pkl `
  --metrics metrics.json `
  python src/model/model_evaluation.py
# check dvc.yaml file and metrics.json
dvc repro # check dvc.lock
dvc dag
dvc metrics show
git add .
git commit -m "Add DVC pipeline"
```
- Run the following commands
```sh
dvc exp run # open DVC extension and check
dvc exp show

# change the params.yaml

dvc exp remove --queue # remove queue if any

  -S data_collection.test_size=0.15,0.45,0.55 `
  -S model_building.n_estimators=550
dvc exp run --run-all
dvc exp show

dvc exp run --queue `
 -S data_collection.test_size=0.15,0.45,0.55 `
 -S model_building.n_estimators=850,650,750
dvc exp run --run-all

dvc exp apply tough-tosh # check water-potability-tracking\params.yaml

```
